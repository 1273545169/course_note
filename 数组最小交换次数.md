# 数组最小交换次数
import sys


def minSwap(nums):
    mp = {}
    sorted_nums = sorted(nums)
    for i in range(len(nums)):
        mp[sorted_nums[i]] = i

    loops = 0
    flags = []
    for i in range(len(nums)):
        flags.append(False)
    for i in range(len(nums)):
        if not flags[i]:
            j = i
            while not flags[j]:
                flags[j] = True
                j = mp[nums[j]]
            loops += 1
    return len(nums) - loops


n = int(sys.stdin.readline().strip())

nums = []
for i in range(n):
    nums += list(map(int, sys.stdin.readline().strip().split()))

print(minSwap(nums))
